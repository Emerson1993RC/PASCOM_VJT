function getHeadColor(ev) {
  const t = ev.title.toLowerCase();
  if(t.includes('quinta')) return '#5b2d8e';
  if(t.includes('manhã')) return '#e67e22';
  if(t.includes('noite')) return '#2c3e6b';
  if(t.includes('sexta')) return '#1a6b4a';
  if(t.includes('corpus')) return '#8e2d2d';
  return '#2d5a8e';
}

// CSV PARSE

function parseCSV(text){
  const rows=[];let i=0;
  while(i<text.length){
    const row=[];
    while(i<text.length&&text[i]!=='\n'){
      if(text[i]==='"'){
        let f='';i++;
        while(i<text.length){
          if(text[i]==='"'&&text[i+1]==='"'){f+='"';i+=2;}
          else if(text[i]==='"'){i++;break;}
          else{f+=text[i];i++;}
        }
        row.push(f);if(text[i]===',')i++;
      }else{
        let f='';
        while(i<text.length&&text[i]!==','&&text[i]!=='\n'){f+=text[i];i++;}
        row.push(f);if(text[i]===',')i++;
      }
    }
    if(text[i]==='\n')i++;
    if(row.length>0&&row.some(c=>c.trim()))rows.push(row);
  }
  return rows;
}

function extractDate(col){
  const m=col.match(/\[(\d{2})\/(\d{2})\/(\d{4})/);
  if(!m)return null;
  const day=parseInt(m[1]),month=parseInt(m[2])-1,year=parseInt(m[3]);
  if(day===0)return null;
  return{day,month,year};
}
function extractTime(col){
  const ps=[/Chegada[^\d]*(\d{1,2}:\d{2})hs/i,/Chegada[^\d]*(\d{1,2})hs/i,/(\d{1,2}:\d{2})hs/i,/(\d{1,2})hs/i];
  for(const p of ps){const m=col.match(p);if(m)return m[1].includes(':')?m[1]:m[1]+':00';}
  return null;
}
function extractTitle(col){
  let t=col.split('\n')[0].trim().replace(/\s*\[.*?\]\s*/g,'').trim();
  t=t.replace(/Escala para as quartas-feiras\.\s*/i,'📷 Quarta-feira')
     .replace(/Escala para as quintas-feiras\.\s*/i,'📷 Quinta-feira')
     .replace(/Escala para as sextas-feiras\.\s*/i,'📷 Sexta-feira')
     .replace(/Escala para os Domingos \(Manhã\)\.\s*/i,'🌅 Domingo (Manhã)')
     .replace(/Escala para os Domingos \(Noite\)\.\s*/i,'🌙 Domingo (Noite)')
     .replace(/Operador\(a\) de Câmeras:.*$/i,'')
     .replace(/Corpus Christi\s*/i,'✝️ Corpus Christi')
     .replace(/Dia "D Missionário".*$/i,'🌍 Dia D Missionário')
     .replace(/CPP\s*/i,'🙏 CPP').trim();
  const ib=col.match(/\[[\d/]+\s+\(([^)]+)\)\]/);
  if(ib)t+=' — '+ib[1];
  return t||'Evento';
}

function processData(csvText){
  const rows=parseCSV(csvText);
  if(rows.length<2){alert('Arquivo CSV inválido.');return;}
  const headers=rows[0];
  const nameColIdx=headers.findIndex(h=>h.trim().toLowerCase()==='nome');
  if(nameColIdx<0){alert('Coluna "Nome" não encontrada.');return;}
  nameColors={};let ci=0;
  for(let r=1;r<rows.length;r++){
    const n=(rows[r][nameColIdx]||'').trim();
    if(n&&!nameColors[n]){nameColors[n]=ci%10;ci++;}
  }
  allEvents=[];
  for(let c=0;c<headers.length;c++){
    if(c===nameColIdx)continue;
    const col=headers[c];
    const di=extractDate(col);if(!di)continue;
    const names=[];
    for(let r=1;r<rows.length;r++){
      const val=(rows[r][c]||'').trim();
      if(val.toLowerCase()==='disponível'){
        const n=(rows[r][nameColIdx]||'').trim();
        if(n)names.push(n);
      }
    }
    if(names.length===0)continue;
    allEvents.push({...di,title:extractTitle(col),time:extractTime(col),names,colHeader:col});
  }
  allEvents.sort((a,b)=>{
    const da=new Date(a.year,a.month,a.day),db=new Date(b.year,b.month,b.day);
    return da-db||(a.time||'00:00').localeCompare(b.time||'00:00');
  });
  availableMonths=[];const seen=new Set();
  for(const ev of allEvents){
    const k=`${ev.year}-${ev.month}`;
    if(!seen.has(k)){seen.add(k);availableMonths.push({year:ev.year,month:ev.month});}
  }
  if(availableMonths.length>0){currentYear=availableMonths[0].year;currentMonth=availableMonths[0].month;}
  selectedDay=null;
  buildUI();
}

function buildCalendar(){
  document.getElementById('monthLabel').textContent=PT_MONTHS[currentMonth]+' '+currentYear;
  const grid=document.getElementById('calendarGrid');
  grid.innerHTML='';
  for(const d of PT_DAYS){const el=document.createElement('div');el.className='day-label';el.textContent=d;grid.appendChild(el);}
  const firstDay=new Date(currentYear,currentMonth,1).getDay();
  const dim=new Date(currentYear,currentMonth+1,0).getDate();
  const evByDay={};
  for(const ev of allEvents){
    if(ev.year===currentYear&&ev.month===currentMonth){
      evByDay[ev.day]=(evByDay[ev.day]||0)+1;
    }
  }
  const today=new Date();
  for(let i=0;i<firstDay;i++){const el=document.createElement('div');el.className='cal-day empty';grid.appendChild(el);}
  for(let d=1;d<=dim;d++){
    const el=document.createElement('div');el.className='cal-day';
    if(today.getDate()===d&&today.getMonth()===currentMonth&&today.getFullYear()===currentYear)el.classList.add('today');
    const cnt=evByDay[d]||0;
    if(cnt>0){el.classList.add('has-event');el.onclick=()=>selectDay(d);}
    if(selectedDay&&selectedDay.day===d&&selectedDay.month===currentMonth&&selectedDay.year===currentYear)el.classList.add('selected');
    const numEl=document.createElement('div');numEl.className='day-num';numEl.textContent=d;el.appendChild(numEl);
    if(cnt>0){
      const dot=document.createElement('div');dot.className='event-dot';el.appendChild(dot);
      if(cnt>1){const c=document.createElement('div');c.className='event-count';c.textContent=cnt+'x';el.appendChild(c);}
    }
    grid.appendChild(el);
  }
}

function launchDayWizard(){
  if(!selectedDay) return;
  const dayEvents = allEvents.filter(e=>e.day===selectedDay.day&&e.month===selectedDay.month&&e.year===selectedDay.year);
  if(dayEvents.length===0) return;

  wizardEvents = dayEvents;
  wizardData = {};
  wizardQueue = [];

  dayEvents.forEach((ev, evIdx)=>{
    wizardData[evIdx] = {};
    SLOTS.forEach(slot => wizardQueue.push({ev, evIdx, slot}));
  });

  wizardQueueIdx = 0;
  document.getElementById('wizardModal').classList.remove('hidden');
  renderWizardStep();
}

// Keep old startWizard for long-press (single event)
function startWizard(e, nome){
  const card = e.target ? e.target.closest('.event-card') : e.closest('.event-card');
  if(!card) return;
  const key = card.dataset.evkey;
  const ev = allEvents.find(e=>`${e.day}-${e.month}-${e.year}-${e.time||''}`===key);
  if(!ev) return;

  wizardEvents = [ev];
  wizardData = {0:{}};
  wizardQueue = SLOTS.map(slot=>({ev, evIdx:0, slot}));
  wizardQueueIdx = 0;
  document.getElementById('wizardModal').classList.remove('hidden');
  renderWizardStep();
}

function renderWizardStep(){
  if(wizardQueueIdx >= wizardQueue.length){
    closeWizard();
    renderPreview();
    return;
  }
  const step = wizardQueue[wizardQueueIdx];
  const totalSteps = wizardQueue.length;
  const slotIdx = SLOTS.indexOf(step.slot);
  const evTotal = wizardEvents.length;
  const evNum = step.evIdx + 1;

  // Progress: show SLOTS dots, reset per event
  const bar = document.getElementById('progressBar');
  bar.innerHTML = SLOTS.map((_,i)=>{
    const cls = i < slotIdx ? 'done' : i === slotIdx ? 'active' : '';
    return `<div class="progress-dot ${cls}"></div>`;
  }).join('');

  document.getElementById('modalStepLabel').textContent =
    `Evento ${evNum} de ${evTotal} · Cargo ${slotIdx+1} de ${SLOTS.length}`;
  document.getElementById('modalQuestion').textContent = step.slot;

  // Title cleaned
  const titleClean = step.ev.title.replace(/📷|🌅|🌙|✝️|🌍|🙏|📸/g,'').trim();
  document.getElementById('modalPerson').innerHTML =
    `Para: <strong style="color:var(--gold-light)">${titleClean}</strong><br>
     <small style="color:var(--text-muted)">${String(step.ev.day).padStart(2,'0')}/${String(step.ev.month+1).padStart(2,'0')} ${step.ev.time?'— '+step.ev.time+'hs':''}</small>`;

  // Name chips
  const avail = step.ev.names;
  let chips = '<div style="display:flex;flex-wrap:wrap;gap:7px;margin-top:12px;">';
  chips += avail.map(n=>`<button
    onclick="wizardPickName('${n.replace(/'/g,"\\'")}',this)"
    style="padding:5px 13px;border-radius:20px;border:1px solid rgba(201,168,76,.4);background:rgba(201,168,76,.1);color:#f0d080;font-size:.76rem;font-weight:700;cursor:pointer;transition:all .2s;"
    onmouseover="this.style.background='rgba(201,168,76,.28)'"
    onmouseout="this.style.background=this.dataset.sel?'rgba(201,168,76,.4)':'rgba(201,168,76,.1)'"
    >${n.trim()}</button>`).join('');
  chips += '</div>';
  document.getElementById('modalPerson').innerHTML += chips;

  // Pre-highlight if already picked
  const already = wizardData[step.evIdx][step.slot];
  if(already && already !== '—'){
    setTimeout(()=>{
      document.querySelectorAll('#modalPerson button').forEach(b=>{
        if(b.textContent.trim()===already.trim()){
          b.style.background='rgba(201,168,76,.4)';
          b.style.outline='2px solid #f0d080';
          b.dataset.sel='1';
        }
      });
    },10);
  }
}

function wizardPickName(name, btn){
  btn.parentElement.querySelectorAll('button').forEach(b=>{
    b.style.outline='none'; b.style.background='rgba(201,168,76,.1)'; delete b.dataset.sel;
  });
  btn.style.outline='2px solid #f0d080';
  btn.style.background='rgba(201,168,76,.35)';
  btn.dataset.sel='1';
  const step = wizardQueue[wizardQueueIdx];
  wizardData[step.evIdx][step.slot] = name;
}

function wizardAnswer(yes){
  const step = wizardQueue[wizardQueueIdx];
  if(yes){
    if(!wizardData[step.evIdx][step.slot])
      wizardData[step.evIdx][step.slot] = step.ev.names[0]||'—';
  } else {
    wizardData[step.evIdx][step.slot] = '—';
  }
  wizardQueueIdx++;
  renderWizardStep();
}

function closeWizard(){
  document.getElementById('wizardModal').classList.add('hidden');
}

let toastTimer=null;
function copyName(name,el){
  const doToast=()=>{
    document.querySelectorAll('.name-tag.copied').forEach(t=>t.classList.remove('copied'));
    el.classList.add('copied');
    setTimeout(()=>el.classList.remove('copied'),1200);
    const toast=document.getElementById('copyToast');
    document.getElementById('toastName').textContent=name;
    toast.classList.add('show');
    clearTimeout(toastTimer);
    toastTimer=setTimeout(()=>toast.classList.remove('show'),2200);
  };
  if(navigator.clipboard){navigator.clipboard.writeText(name).then(doToast).catch(doToast);}
  else{
    const ta=document.createElement('textarea');ta.value=name;ta.style.cssText='position:fixed;opacity:0';
    document.body.appendChild(ta);ta.select();document.execCommand('copy');document.body.removeChild(ta);
    doToast();
  }

function navMonth(delta){
  currentMonth+=delta;
  if(currentMonth>11){currentMonth=0;currentYear++;}
  if(currentMonth<0){currentMonth=11;currentYear--;}
  selectedDay=null;buildUI();
}
document.getElementById('prevMonth').onclick=()=>navMonth(-1);
document.getElementById('nextMonth').onclick=()=>navMonth(1);

  document.getElementById('calendarSection').classList.add('visible');
  buildCalendar();
  buildSchedule();
}

// ── File Upload ──────────────────────────────────────────────
function handleFile(file){
  if(!file||!file.name.endsWith('.csv')){alert('Selecione um arquivo .csv');return;}
  const reader=new FileReader();
  reader.onload=e=>{
    document.getElementById('fileStatus').innerHTML=`<div class="file-loaded">✅ ${file.name} carregado</div>`;
    processData(e.target.result);
  };
  reader.readAsText(file,'UTF-8');
}
document.getElementById('fileInput').addEventListener('change',e=>{if(e.target.files[0])handleFile(e.target.files[0]);});
const zone=document.getElementById('uploadZone');
zone.addEventListener('dragover',e=>{e.preventDefault();zone.classList.add('drag-over');});
zone.addEventListener('dragleave',()=>zone.classList.remove('drag-over'));
zone.addEventListener('drop',e=>{e.preventDefault();zone.classList.remove('drag-over');if(e.dataTransfer.files[0])handleFile(e.dataTransfer.files[0]);});
